 CREATE TABLE match_deliveries_2024 (
         match_id INT,
         season INT,
         match_no VARCHAR(10),
         date DATE,
         venue VARCHAR(100),
         batting_team VARCHAR(50),
         bowling_team VARCHAR(50),
         innings INT,
         `over` INT,
         striker VARCHAR(50),
         non_striker VARCHAR(50),
         bowler VARCHAR(50),
         runs_of_bat INT,
         extras INT,
         wide INT,
         legbyes INT,
         byes INT,
         noballs INT,
         wicket_type VARCHAR(30),
         player_dismissed VARCHAR(50),
         fielder VARCHAR(50)
       );

 CREATE TABLE ipl_matches_2024 (
         match_id INT PRIMARY KEY,
         season INT,
         date DATE,
         match_no VARCHAR(20),
         venue VARCHAR(100),
         city VARCHAR(50),
         team1 VARCHAR(50),
         team2 VARCHAR(50),
         toss_winner VARCHAR(50),
         toss_decision VARCHAR(10),
         innings1_score VARCHAR(10),
         innings1_wickets INT,
         innings2_score VARCHAR(10),
         innings2_wickets INT,
         winning_team VARCHAR(50),
         margin VARCHAR(20),
         won_by VARCHAR(20),
         player_of_the_match VARCHAR(50),
         team1_players TEXT,
         team2_players TEXT,
         umpire1 VARCHAR(50),
         umpire2 VARCHAR(50),
         umpire3 VARCHAR(50)
       );
